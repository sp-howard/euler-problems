Multiples of 3 and 5

Input a positive integer, and this Javascript function will return the sum of all multiples of 3 and 5 up to that number.


TEST
CODE
  var multiple = function(num) {
  var sum = 0;

  for (i = 0; i < num; i++) {
    if (i % 5 === 0 || i % 3 === 0) {
      sum += i
    }
  }

  return sum;
}

var fib = function(num) {
  var sum = 0,
      first = 0,
      second = 1,
      even = 0;

  while (sum < num) {
    sum = first + second;
    first = second;
    second = sum
    if (sum % 2 === 0) {
      even += sum;
    }
  };

  return even;
}
